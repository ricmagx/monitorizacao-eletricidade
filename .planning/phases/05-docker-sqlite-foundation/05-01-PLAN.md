2. Create `Dockerfile`:
```dockerfile
FROM python:3.11-slim
WORKDIR /app
COPY requirements-docker.txt .
RUN pip install --no-cache-dir -r requirements-docker.txt
COPY . .
RUN chmod +x entrypoint.sh
HEALTHCHECK --interval=30s --timeout=3s --start-period=10s \
  CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health')" || exit 1
CMD ["./entrypoint.sh"]
```

3. Create `entrypoint.sh`:
```bash
#!/bin/sh
set -e
cd /app
alembic upgrade head
exec uvicorn src.web.app:app --host 0.0.0.0 --port 8000
```

4. Create `docker-compose.yml` (no `version` key — Docker Compose v2 ignores it):
```yaml
services:
  energia:
    build: .
    ports:
      - "8000:8000"
    volumes:
      - energia_data:/app/data
    environment:
      - DB_PATH=/app/data/energia.db
    restart: unless-stopped

volumes:
  energia_data:
```

5. Create `.env.example`:
```
DB_PATH=/app/data/energia.db
APP_PORT=8000
```

6. Append to `.gitignore` (if not already present):
```

```
