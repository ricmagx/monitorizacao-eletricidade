    ```python
    def load_custos_reais(custos_path: Path) -> dict[str, float]:
        """Le custos_reais.json e retorna entries dict. {} se nao existe."""
        if not custos_path.exists():
            return {}
        data = json.loads(custos_path.read_text(encoding="utf-8"))
        return data.get("entries", {})

    def save_custo_real(custos_path: Path, year_month: str, custo_eur: float) -> None:
        """Adiciona/actualiza entry em custos_reais.json. Cria ficheiro se nao existe."""
        entries = load_custos_reais(custos_path)
        entries[year_month] = custo_eur
        custos_path.parent.mkdir(parents=True, exist_ok=True)
        custos_path.write_text(
            json.dumps({"updated_at": datetime.now().isoformat(), "entries": entries},
                       indent=2, ensure_ascii=False),
            encoding="utf-8"
        )

    def build_custo_chart_data(
        consumo_data: list[dict],
        analysis: dict | None,
        custos_reais: dict[str, float]
    ) -> dict:
        """Constroi dados para o grafico de custo misto (bar estimativa + line custo real).

        - labels: lista de year_month de consumo_data
        - estimativa_data: custo do fornecedor actual por mes (de analysis.history[].current_supplier_result.total_eur)
        - custo_real_data: custo real por mes (de custos_reais), None para meses sem dado

        Per D-07: meses sem custo real = None (omitidos na linha, nao zero).
        """
        labels = [row["year_month"] for row in consumo_data]

        # Estimativa do fornecedor actual por mes
        estimativa_by_month = {}
        if analysis and "history" in analysis:
            for entry in analysis["history"]:
                ym = entry.get("year_month")
                csr = entry.get("current_supplier_result", {})
                if ym and "total_eur" in csr:
                    estimativa_by_month[ym] = round(csr["total_eur"], 2)

        estimativa_data = [estimativa_by_month.get(ym) for ym in labels]
        custo_real_data = [custos_reais.get(ym) for ym in labels]  # None se nao existe

        return {"labels": labels, "estimativa_data": estimativa_data, "custo_real_data": custo_real_data}
    ```

    **2. Criar `src/web/routes/custos_reais.py`:**

    ```python
    from fastapi import APIRouter, Request, Form
    from fastapi.responses import HTMLResponse
    from pathlib import Path
    from src.web.services.data_loader import save_custo_real, load_custos_reais, load_consumo_csv, load_analysis_json, build_custo_chart_data, load_locations

    router = APIRouter()

    @router.post("/local/{local_id}/custo-real", response_class=HTMLResponse)
    async def post_custo_real(
        request: Request,
        local_id: str,
        year_month: str = Form(...),
        custo_eur: float = Form(...),
    ):
        project_root = request.app.state.project_root
        custos_path = project_root / "data" / local_id / "custos_reais.json"
        save_custo_real(custos_path, year_month, custo_eur)

        # Retornar fragmento actualizado do grafico de custo
        # (re-render custo_chart.html + custo_form.html com dados actualizados)
        locations = load_locations(request.app.state.config_path)
        location = next((l for l in locations if l["id"] == local_id), None)
        if not location:
            return HTMLResponse(status_code=404, content="Local nao encontrado")

        consumo_data = load_consumo_csv(project_root / location["pipeline"]["processed_csv_path"])
        analysis = load_analysis_json(project_root / location["pipeline"]["analysis_json_path"])
        custos_reais = load_custos_reais(custos_path)
        custo_chart = build_custo_chart_data(consumo_data, analysis, custos_reais)

        templates = request.app.state.templates  # set in app.py (step 3)
        return templates.TemplateResponse(
            "partials/custo_section.html",
            {"request": request, "custo_chart": custo_chart, "custos_reais": custos_reais,
             "consumo_data": consumo_data, "selected_location": location}
        )
    ```
